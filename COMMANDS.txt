DEPLOYMENT FOR PRODUCTION (optional):
================================================================================

# Create systemd service (Linux):
# Create /etc/systemd/system/openclaw-pipeline.service with:
[Unit]
Description=OpenClaw Pipeline Service
After=network.target

[Service]
Type=simple
User=beakthor0301
WorkingDirectory=/home/beakthor0301/Code/openclaw-playground
ExecStart=/usr/bin/python3 scripts/run-pipeline.py
Environment="PATH=/home/beakthor0301/Code/openclaw-playground/venv/bin"

[Install]
WantedBy=multi-user.target

# Then enable and start:
sudo systemctl enable openclaw-pipeline
sudo systemctl start openclaw-pipeline
